Code

#include <cstdio>
typedef long long ll
const int MAX_N=25;
int p[MAX_N],pn;
int phi[MAX_N];
bool v[MAX_N];
int n;
ll qkpow(ll a,int b){ ll r=1; for(; b; a=a*a,b>>=1) if(b&1) r=r*a; return r; }
void euler_sieve(int n=23){
	phi[1]=1;
	for(int i=2; i<=n; ++i){
		if(!v[i]) p[++pn]=i,phi[i]=i-1;
		for(int j=1; i*p[j]<=n; ++j){
			v[i*p[j]]=1;
			if(i%p[j]){ phi[i*p[j]]=phi[i]*(p[j]-1); }
			else{ phi[i*p[j]]=phi[i]*p[j]; break; }
		}
	}
}
int main(){
	euler_sieve();
	for(scanf("%d",&n); ~n; scanf("%d",&n)){
		if(!n){ puts("0"); continue; } 
		ll ans=0;
		for(int i=1; i*i<=n; ++i) if(n%i==0){
			ans+=qkpow(3,i)*phi[n/i];
			if(i*i!=n) ans+=qkpow(3,n/i)*phi[i];
		}
		if(n&1) ans+=n*qkpow(3,(n>>1)+1);
		else ans+=(n>>1)*(qkpow(3,n>>1)+qkpow(3,(n>>1)+1));
		ans/=(n<<1);
		printf("%lld\n",ans); 
	}
	return 0;
}
