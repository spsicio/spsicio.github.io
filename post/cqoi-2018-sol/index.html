D1T1 破解 D-H 协议

Solution

  • 只需要实现 同余取对数 就可解决问题。

Code

int g,p,t,n;
#define mul(a,b) (1ll*(a)*(b)%p)
int qkpow(int a,int b){
	int r=1;
	for(; b; a=mul(a,a),b>>=1) if(b&1) r=mul(r,a);
	return r;
}
int bsgs(int b){
	static std::unordered_map<int,int> mp;
	mp.clear();
	int vt=1;
	for(int j=1; j<=t; ++j) mp[mul(b,vt=mul(vt,g))]=j;
	for(int i=1,j,v=vt; i<=t; ++i,v=mul(v,vt)) if((j=mp[v])) return i*t-j;
	return -1;
}
int main(){
	rd(g,p,n); t=ceil(sqrt(p));
	for(int i=1,A,B; i<=n; ++i){ rd(A,B); wr(qkpow(A,bsgs(B)),'\n'); }
	return 0;
}

Solution

  • 若 vvv 可以看到 uuu 的消息，就连一条 (u,v)(u,v)(u,v) 的有向边。答案为原图以 111 为根的叶向树形图个数。

const int MAX_N=255,MOD=10007;
typedef int mat[MAX_N][MAX_N];
inl void inc(int &a){ a=(a==MOD-1? 0: a+1); }
inl void dec(int &a){ a=(a==0? MOD-1: a-1); }
inl void subeq(int &a,int b){ a-=b; if(a<0) a+=MOD; }
#define mul(a,b) (1ll*(a)*(b)%MOD)
mat A; int n,m;
int det1(mat A,int n){
	int res=1;
	for(int i=2; i<=n; ++i){
		for(int j=i+1; j<=n; ++j) while(A[j][i]){
			int t=A[i][i]/A[j][i];
			for(int k=i; k<=n; ++k) subeq(A[i][k],mul(t,A[j][k]));
			std::swap(A[i],A[j]);
			res=MOD-res;
		}
		res=mul(res,A[i][i]);
	}
	return res;
}
int main(){
	rd(n,m);
	for(int u,v; m; --m){ rd(v,u); inc(A[v][v]); dec(A[u][v]); }
	wr(det1(A,n),'\n');
	return 0;
}

  • 不妨设 f(i,j,0/1)f(i,j,0/1)f(i,j,0/1) 表示长为 iii 且以 0/10/10/1 结尾的合法序列对 jjj 次项的贡献。
    • 显然有 f(i,j,0)=f(i−1,j,0)+f(i−1,j,1)f(i,j,0)=f(i-1,j,0)+f(i-1,j,1)f(i,j,0)=f(i−1,j,0)+f(i−1,j,1)。
    • 考虑每个合法方案的贡献，求和提公因式可得，f(i,j,1)=∑k=0jCjkf(i−1,k,0)f(i,j,1)=\sum\limits _{k=0}^{j}{C_{j}^{k} f(i-1,k,0)}f(i,j,1)=k=0∑j​Cjk​f(i−1,k,0)。
  • 可以使用矩乘快速幂解决。

Attention

  • 有点卡常。
    • 矩乘时每个点值先放在 long long 里储存起来，最后一步再取模。
    • 转移矩阵总分为四个上三角矩阵，可以特化，消除一半运算量。（这里的代码没有这样做。）

const int MAX_N=10000005,MAX_A=47,MAX_T=95,MAX_U=185;
int n,a,b,m,t,u;
int C[MAX_T][MAX_T];
inl int add(int a,int b){ a+=b; return a>=m? a-m: a; } 
inl int sub(int a,int b){ a-=b; return a<0? a+m: a; }
inl void addeq(int &a,int b){ a+=b; if(a>=m) a-=m; }
#define mul(a,b) (1ll*(a)*(b)%m)
struct Mat{
	int r,c,v[MAX_U][MAX_U];
	Mat(int r=0,int c=0):r(r),c(c){ memset(v,0,sizeof(v)); } 
	int* operator[](int i){ return v[i]; } 
	Mat operator*(Mat B){
		Mat R(r,B.c);
		for(int i=0; i<r; ++i)
		for(int j=0; j<R.c; ++j){
			ll res=0;
			for(int k=0; k<c; ++k) res+=1ll*v[i][k]*B[k][j];
			R[i][j]=res%m;
		}
		return R;
	}
}F,A;
void qkpow(Mat &F,Mat A,int b){ for(; b; A=A*A,b>>=1) if(b&1) F=F*A; }
int main(){
	rd(n,a,b,m); t=a+b+1,u=t<<1;
	for(int i=0; i<=t; ++i){
		C[i][0]=1;
		for(int j=1; j<=i; ++j)
			C[i][j]=add(C[i-1][j-1],C[i-1][j]);
	}
	F=Mat(1,u),A=Mat(u,u);
	F[0][0]=1; for(int j=t; j<u; ++j) F[0][j]=1;
	for(int j=0; j<t; ++j){
		A[j][j]=A[j+t][j]=1;
		for(int k=0; k<=j; ++k) A[k][j+t]=C[j][k];
	}
	qkpow(F,A,n-1);
	int ans=0;
	for(int i=0,v=1; i<=a; ++i,v=mul(v,n)){
		int val=mul(C[a][i],v);
		if((a-i)&1) val=m-val;
		addeq(ans,mul(val,add(F[0][a+b-i],F[0][a+b-i+t])));
	}
	wr(ans,'\n');
	return 0;
}

D2T1 解锁屏幕

Solution

  • 先考虑搜索做法：尝试从每个点当作起点，进行搜索，合法的转移需要满足条件：①不能到达已经经过的点；②路径上不能有没有经过的点。因此我们需要在预处理出任意两点间路径的点集，并且在搜索的过程中记录经历过的点集。可以使用 状态压缩 完成。
  • 设 f(i,S)f(i,S)f(i,S) 为当前在 iii 点，经历过的点集为 SSS 的方案总数。就可将搜索优化为 DP。

Attention

  • 有点小卡常。
    • 使用数组记录集合的大小。
    • 将点从 0 索引在每次访问集合元素时减少一次减法操作。

	return (a.x-c.x)*(b.y-c.y)==(b.x-c.x)*(a.y-c.y)
		and ((a.x!=b.x)? ((a.x>c.x)!=(b.x>c.x)): ((a.y>c.y)!=(b.y>c.y)));
}
int main(){
	rd(n); u=1<<n;
	for(int i=0; i<n; ++i) rd(p[i].x,p[i].y);

		for(int k=0; k<n; ++k) if(k!=i and k!=j and iscatch(p[i],p[j],p[k]))
			r[i][j]|=(1<<k);
		r[j][i]=r[i][j];
	}
	for(int i=0; i<n; ++i) f[(1<<i)][i]=1;
	for(int s=0; s<u; ++s)
	for(int i=0; i<n; ++i) if(s&(1<<i)){
		for(int j=0; j<n; ++j) if(!(s&(1<<j)) and (r[i][j]|s)==s)
			addeq(f[s|(1<<j)][j],f[s][i]);
	}
	int ans=0;
	for(int s=0; s<u; ++s) if((popcount[s]=popcount[s>>1]+(s&1))>=4)
	for(int i=0; i<n; ++i) if(s&(1<<i)) addeq(ans,f[s][i]);
	wr(ans,'\n');
	return 0;
}

  • 不妨设 f(i)f(i)f(i) 表示取下 iii 连环的最少步数。

      • 要想拆掉最左边那个环，一定要先形成 1100⋯001100 \cdots 001100⋯00 的状态才能拆掉。共需要 f(i−2)+1f(i-2)+1f(i−2)+1 步。

const int MAX_W=7535;
struct BigInt{
	enum{BASE=100000000};
	int len; ll v[MAX_W];
	friend void wr(BigInt x){
		for(int i=x.len; i; --i){
			for(int lim=BASE/10; i!=x.len and lim>1 and  x.v[i]<lim; lim/=10)
				wr('0');
			wr(x.v[i]);
		}
	}
	inl void clz(){ while(len>1 and !v[len]) --len; }
	BigInt(int b=0){
		len=0; memset(v,0,sizeof(v));
		do v[++len]=b%BASE; while(b/=BASE);
	}
	BigInt operator*(BigInt b){
		BigInt r; r.len=len+b.len-1;
		for(int i=1; i<=len; ++i)
		for(int j=1; j<=b.len; ++j){
			r.v[i+j-1]+=v[i]*b.v[j];
			r.v[i+j]+=r.v[i+j-1]/BASE;
			r.v[i+j-1]%=BASE;
		}
		if(r.v[r.len+1]) ++r.len;
		r.clz(); 
		return r;
	}
	BigInt operator/(int b){
		BigInt r; r.len=len;
		int g=0;
		for(int i=len; i; --i){
			g=g*BASE+v[i];
			r.v[i]=g/b;
			g-=r.v[i]*b;
		}
		r.clz();
		return r;
	}
}A=2;

	BigInt r=1;
	for(; b; a=a*a,b>>=1) if(b&1) r=r*a;
	return r;
}

int main(){
	for(rd(m); m; --m){ rd(n); wr(qkpow(2,n+1)/3,'\n'); }

D2T3 异或序列

Solution

  • 求任意区间的异或和，先对原序列求前缀异或和，使 gi=a1⊕a2⊕⋯⊕aig_{i}=a_{1} \oplus a_{2} \oplus \cdots \oplus a_{i}gi​=a1​⊕a2​⊕⋯⊕ai​。
  • 对于区间 [L,R][L, R][L,R] 的询问，结果为 ∑i=LR∑j=Li[gj−1=k⊕gi)]\sum\limits_{i=L}^{R}{\sum\limits _{j=L}^{i}{[g_{j-1}=k \oplus g_{i})]}}i=L∑R​j=L∑i​[gj−1​=k⊕gi​)]。
    • 可以记录下区间内每种权值的数量将后一个求和的复杂度优化掉，一个询问的复杂度为 O(n)O(n)O(n) 。
  • 使用 莫队算法 可以优化掉大量询问间重复的操作，解决该问题。

const int MAX_N=100005,MAX_M=100005,MAX_V=100005;
int len;
struct QryData{
	int l,r,id;
	QryData(int l=0,int r=0,int id=0):l(l),r(r),id(id){}
	bool operator<(const QryData &b)const{
		return (l/len!=b.l/len)? (l<b.l): (((l/len)&1)? (r<b.r): (r>b.r));
	}
}q[MAX_M];
int cur_l,cur_r,cur_ans;
int ans[MAX_M];
int cnt[MAX_V];
int g[MAX_N];
int n,m,k;
void inc(int p){ cur_ans+=cnt[k^g[p]]; ++cnt[g[p]]; }
void dec(int p){ --cnt[g[p]]; cur_ans-=cnt[k^g[p]]; }
int main(){
	rd(n,m,k); len=sqrt(n);
	for(int i=1; i<=n; ++i) rd(g[i]),g[i]^=g[i-1];
	for(int i=1,l,r; i<=m; ++i){ rd(l,r); q[i]=QryData(l-1,r,i); }
	std::sort(q+1,q+m+1);
	++cnt[0],++cnt[g[1]];
	cur_l=0,cur_r=1,cur_ans=(g[1]==k);
	for(int i=1; i<=m; ++i){
		while(cur_r<q[i].r) inc(++cur_r);
		while(cur_r>q[i].r) dec(cur_r--);
		while(cur_l>q[i].l) inc(--cur_l);
		while(cur_l<q[i].l) dec(cur_l++);
		ans[q[i].id]=cur_ans;
	}
	for(int i=1; i<=m; ++i) wr(ans[i],'\n');
	return 0;
}
