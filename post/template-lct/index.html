Link-Cut Tree 是一种基于实链剖分，支持添加与删除边，利用辅助树维护森林信息的数据结构。

实链剖分

  • 对于每个结点与儿子间的边，可选取一条边作为实边，其余边为虚边。
  • 由实边组成的链为实链。

辅助树

  • 原树中的结点与辅助树中的结点一一对应。
  • 原树中的实边：将连接的两个点加入到同一颗 Splay 中，要求 Splay 中序遍历得到的顺序为深度由小到大的顺序。（每棵 Splay 维护了一条实链）
  • 原树中的虚边：将深度较深点所在 Splay 根结点的父节点设为深度较浅点。

常规操作

Splay 操作

与普通 Splay 写法稍有不同。

#define nroot(x) (ch[fa[x]][0]!=x and ch[fa[x][1]]!=x)
#define isrc(x) (fa[x][1]==x)
void push_up(int x);
void push_dn(int x);
void rotate(int x){
    int y=fa[x],z=fa[y],k=isrc(x),w=ch[x][!k];
    if(nroot(y)) ch[z][isrc(y)]=x; ch[x][!k]=y; ch[y][k]=w;
    if(w) fa[w]=y; fa[y]=x; fa[x]=z;
    push_up(y);
}
void splay(int x){
    static int s[MAX_N],top;
    for(int y=x; nroot(y); y=fa[y]) s[++top]=y;
    while(top) push_dn(s[top--]);
    while(nroot(x)){
        int y=fa[x];
        if(nroot(y)) rotate((isrc(x)^isrc(y))? x: y);
        rotate(x);
    }
    push_up(x);
}

新操作

int access(int x){
    int y=0;
    for(; x; x=fa[y=x]) splay(x),ch[x][1]=y,push_up(x);
    return y;
}
void evert(int x){ push_rev(access(x)); }
int split(int x,int y){ evert(x); return access(y); }
int find(int x){
    for(x=access(x); ch[x][0]; x=ch[x][0]) push_dn(x);
    return splay(x),x; // 需要 splay 保证复杂度。
}
void link(int x,int y){ evert(x),splay(x); fa[x]=y; }
bool chk_link(int x,int y){ evert(x); return find(y)!=x? fa[x]=y: 0; }
void cut(int x,int y){ split(x,y),splay(x); fa[y]=ch[x][0]=0; push_up(x); }
bool chk_cut(int x,int y){
	evert(x);
	if(find(y)!=x or fa[y]!=x or ch[y][0]) return 0;	
	fa[y]=ch[x][1]=0; push_up(x); return 1;
}
