using std::min;
const int MAX_N=5005,MAX_M=2505,MOD=998244353;
int fact[MAX_M],ifact[MAX_M];
int siz[MAX_N],cnt[MAX_N];
int dp[MAX_N][MAX_M];
char typ[MAX_N];
int n,m;
inl void addeq(int &a,int b){ a+=b; if(a>=MOD) a-=MOD; }
#define mul(a,b) (1ll*(a)*(b)%MOD)
#define powt(x) ((x)&1? MOD-1: 1)
struct Grph{
	#define for_edge(g,i,j) for(int i=j; i; i=g.e[i].nxt)
	struct Data{
		int v,nxt;
		Data(int v=0,int nxt=0):v(v),nxt(nxt){}
	}e[MAX_N<<1];
	int head[MAX_N],tot;
	inl void bilnk(int u,int v){
		e[++tot]=Data(v,head[u]); head[u]=tot;
		e[++tot]=Data(u,head[v]); head[v]=tot;
	}
}g;
#define mx_pair(u) (min(siz[u]>>1,cnt[u]))
#define left(u,i) ((typ[u]=='1'? siz[u]-cnt[u]: cnt[u])-(i))
void dfs(int u,int pu=0){
	static int tmp[MAX_M];
	siz[u]=1,cnt[u]=(typ[u]=='1');
	int *dpu=dp[u]; dpu[0]=1;
	for_edge(g,i,g.head[u]){
		int v=g.e[i].v; if(v==pu) continue;
		dfs(v,u);
		int *dpv=dp[v];
		for(int j=mx_pair(u)+mx_pair(v); ~j; --j) tmp[j]=0;
		for(int j=mx_pair(u); ~j; --j) if(dpu[j])
		for(int k=mx_pair(v); ~k; --k) if(dpv[k])
			addeq(tmp[j+k],mul(dpu[j],dpv[k]));
		for(int j=mx_pair(u)+mx_pair(v); ~j; --j) dpu[j]=tmp[j];
		siz[u]+=siz[v],cnt[u]+=cnt[v];
	}
	for(int j=mx_pair(u); j; --j)
		addeq(dpu[j],mul(dpu[j-1],left(u,j-1)));
}
int C(int n,int m){
	return 1ll*fact[n]*ifact[m]%MOD*ifact[n-m]%MOD;
}
int main(){
	rd(n); m=n>>1; rd(typ+1);
	for(int i=1,u,v; i<n; ++i){ rd(u,v); g.bilnk(u,v); }
	dfs(1);
	int *f=dp[1];
	fact[0]=ifact[0]=ifact[1]=1;
	for(int i=1; i<=m; ++i) fact[i]=mul(fact[i-1],i);
	for(int i=2; i<=m; ++i) ifact[i]=mul(MOD-MOD/i,ifact[MOD%i]);
	for(int i=1; i<=m; ++i) ifact[i]=mul(ifact[i-1],ifact[i]);
	for(int i=0; i<=m; ++i) f[i]=mul(f[i],fact[m-i]);
	for(int i=0; i<=m; ++i){
		int ans=0;
		for(int j=i; j<=m; ++j)
			addeq(ans,1ll*C(j,i)*powt(j-i)%MOD*f[j]%MOD);
		wr(ans,'\n');
	}
	return 0;
}
